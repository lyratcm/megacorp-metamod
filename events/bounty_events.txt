namespace = bounty_event
country_event = {
	id = bounty_event.1
	fire_only_once = yes
	trigger = {
		not = {
			has_global_flag = ctf_set
		}	
	}
	hide_window = yes
	immediate = {
		random_galaxy_planet = {
			limit = {
				is_star = yes
				is_star_class = sc_birch
			}
			set_planet_flag = ctf_planet
			set_global_flag = ctf_set
			save_global_event_target_as = ctf_target
		}
		set_global_flag = ESC_AI_research_boost_challenging_scaling
		every_playable_country = {
			country_event = { id = esc_events.5000 days = 0 }
		}
	}
}
country_event = {
	id = bounty_event.2
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		has_country_flag = host
	}
	immediate = {
		if = {
			limit = {
				not = {
					exists = event_target:ctf_target
				}
			}
			random_galaxy_planet = {
				limit = {
					is_star = yes
					is_star_class = sc_birch
				}
				set_planet_flag = ctf_planet
				set_global_flag = ctf_set
				save_global_event_target_as = ctf_target
			}
		}
		event_target:ctf_target = {
			if = {
				limit = {
					solar_system = {
						exists = owner
						owner = {
							is_ai = no
						}
					}
					not = {
						solar_system = {
							any_system_planet = {
								owner = {
									not = {
										is_same_empire = prevprev.owner
									}
								}
							}
						}
					}
				}
				solar_system.owner = {
					add_monthly_resource_mult = {
						resource = unity
						value = 12
					}
				}
			}
			else_if = {
				limit = {
					solar_system = {
						exists = owner
						owner = {
							is_ai = no
						}
					}
				}
				log_error = "not able to give bounty"
			}
		}
	}
}
